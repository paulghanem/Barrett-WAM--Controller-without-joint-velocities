 18function [tau,U1,Ym1,Y1,Yp1,x_22,x_11,P1] = PID(x1,x2,posd,veld,delposd,delveld,fposd,fveld,U,Y,Ym,Yp,P)
 19
 20%-------------------------------------------------------------------------------------
 21%   SETUP THE ARM
 22%-------------------------------------------------------------------------------------    
 23    n = 14;p = 7; q = 14;
 24    
 25 Ts=0.002; 
 26
 27
 28 
 29 
 30
 31
 32yddel=[delposd;delveld];
 33yd = [posd;veld];
 34fyd=[fposd;fveld];
 35
 36
 37 
 38
 39%initial conditions
 40%x1 = [-0.11060;-1.949;-0.1920;3.1840;0.2825;0.09690;-0.27980];
 41%x2 = zeros(7,1);
 42%U = zeros(7,1);
 43Ae = 0.9*eye(n);
 44C= eye(14);
 45Ce = C;
 46Iq = eye(q);
 47Oq = zeros(q);
 48H = [Iq Oq Oq -Ce*Ae;Iq Oq Oq zeros(q,n);Oq Iq Oq zeros(q,n);zeros(n,q) zeros(n,q) zeros(n,q) Ae];
 49Ib = zeros(3*q,3*q+n);
 50for k11 = 1:3*q,Ib(k11,k11) = 1;end
 51sig = 1e-3;
 52sigR = sig;
 53sigQ = 1e-4;%1e-3;
 54alfa = 1e0;
 55beta = 1e-2;
 56p1 = alfa*ones(1,3*q);
 57p2 = beta*ones(1,n);
 58pb = [p1 p2];
 59
 60r1 = 5;
 61r2 = 50;
 62r3 = 1.5;
 63rv = 10;
 64rbb1 = r1*sig^2*ones(1,q/2);
 65rbb2 = r2*0.1*sig^2*ones(1,q/2);
 66rbb12 = [rbb1 rbb2];
 67rbb = [r3*rbb12 rbb12 rbb12];
 68 
 69Rb = 1*diag(rbb);
 70Q = 0*sigQ^2*eye(n);
 71
 72
 73
 74
 75
 76 Be = Ts*[zeros(7,7);inv(D(x1))];
 77    Delyd=fyd-yd;
 78    Del2yd = Delyd*Delyd';
 79   
 80    Gd = [Del2yd Oq Oq zeros(q,n); Oq Oq Oq zeros(q,n); Oq Oq Oq zeros(q,n);zeros(n,q) zeros(n,q) zeros(n,q) Q];
 81   
 82    
 83    
 84   EE = [Ce*Be;zeros(q,p);zeros(q,p);-Be];
 85    K = inv(EE'*EE)*EE'*H*P*Ib'*inv(Ib*P*Ib' + Rb);
 86    PHI = H - EE*K*Ib;
 87    %if max(abs(eig(PHI))) > 1, disp('max(abs(eig(PHI))) > 1'),tk, pause(1),end
 88   
 89    P1 = PHI*P*PHI' + EE*K*Rb*K'*EE' + Gd;
 90    K1 = K(:,1:q);
 91    K2 = K(:,q+1:2*q);
 92    K3 = K(:,2*q+1:3*q);
 93    
 94    
 95      %  E = yd - Y ;
 96       % Ep = ydfut-Yp ;
 97        %U = U + K1*Ep + K2*E;
 98        %Ym = Y;
 99        %Y = Yp;
100         %x_1 = x1 ;
101       
102         %x_11 = x_1; %Since at t=0, velocity is zero
103        %XX2 = zeros(7,1);
104        %x_22= (-inv(D(x_1))*(transpose(G(x_1)))+ inv(D(x_1))*U)*Ts + XX2 ;
105        %Yp = [x_11' x_22']';
106       
107    
108        Em = yddel - Ym ;
109        E = yd - Y ;
110        Ep = fyd-Yp ;
111        U1 = U + K1*Ep + K2*E + K3*Em;
112        Ym1 = Y;
113        Y1 = Yp;
114        
115       
116        
117        
118         
119       Ts=0.002;   
120        % In experimental work you do not add noise instead you only use measurement of position
121        x_11 =  x2*Ts + x1; % Here we are estimating position using the extimate of x2 and measurement of x1 at present time
122        
123        %XX2 = (x_11-x_1)/Ts; 
124        %XX2 = (x1-x1d)/Ts; 
125        
126        %x_22= (-inv(D(pos_meas))*((G(pos_meas)))+ inv(D(pos_meas))*U)*Ts + XX2 ;
127        x3=(-inv(D(x1))*((G(x1)))+ inv(D(x1))*U1);
128       x_22=  x2 + Ts*x3;
129        
130        Yp1 = [x_11' x_22']';
131       
132        tau=U1;
133        
134        
135        
136        
137        
138    
139   
140end
